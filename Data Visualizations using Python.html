# Tree map
import matplotlib
plt.figure(figsize = (20,6))
YEAR = NY_data["County"].astype(str)
VALUES = round(NY_data["Rate"])
# Create a dataset:
my_values=[i**3 for i in range(1,62)]
 
# create a color palette, mapped to these values
cmap = plt.cm.Accent
mini=min(my_values)
maxi=max(my_values)
norm = matplotlib.colors.Normalize(vmin=mini, vmax=maxi)
colors = [cmap(norm(value)) for value in my_values]
squarify.plot(sizes=VALUES, label=YEAR, alpha=0.7, pad = True, color = colors )
plt.title("Unemployment Rates in New York Counties in Dec 2016", size=18)
plt.show();

Bronx = data[data['County'] == 'Bronx County'].groupby('Year').agg('mean')
dfs = pd.concat([Kings, Queens, NewYork, Suffolk, Bronx], axis=1)
